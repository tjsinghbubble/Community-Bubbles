We are working on the rules of the bubbles.

Below is a clean, production-ready but still minimal model for your platform that supports:

App-level rules

Category rules

Community rules

Category hierarchy (Active → Yoga / Running / etc.)

Rule inheritance

Community overrides (future-ready)

The system follows this rule inheritance order:

App Rules
     ↓
Category Rules
     ↓
Community Rules

Overrides work like this:
Global rule
   ↓
Category override
   ↓
Community override


create table rules (
  id uuid primary key default gen_random_uuid(),

  title text not null,
  description text not null,

  rule_level text check (
    rule_level in ('app','category','community')
  ),

  mandatory boolean default false,
  overridable boolean default false,

  created_at timestamp default now()
);

create table category_rules (
  id uuid primary key default gen_random_uuid(),

  category_id uuid references categories(id) on delete cascade,
  rule_id uuid references rules(id) on delete cascade,

  position integer default 0,

  created_at timestamp default now(),

  unique(category_id, rule_id)
);

create table bubble_rules (
  id uuid primary key default gen_random_uuid(),

  bubble_id uuid references communities(id) on delete cascade,
  rule_id uuid references rules(id) on delete cascade,

  position integer default 0,

  created_at timestamp default now(),

  unique(bubble_id, rule_id)
);

create table bubble_rule_overrides (
  id uuid primary key default gen_random_uuid(),

  bubble_id uuid references bubble(id) on delete cascade,
  rule_id uuid references rules(id) on delete cascade,

  override_title text,
  override_description text,

  created_at timestamp default now(),

  unique(bubble_id, rule_id)
);


create table app_rules (
  id uuid primary key default gen_random_uuid(),

  rule_id uuid references rules(id) on delete cascade,

  position integer default 0
);
