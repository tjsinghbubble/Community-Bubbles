Index Naming Convention
Your current pattern is good:


idx_[table]_[column(s)]
Examples:


sql
idx_memberships_user_id
idx_events_bubble_start
idx_notifications_user_created
For composite indexes, list key columns:


sql
idx_[table]_[col1]_[col2]
For partial indexes, add suffix:


sql
idx_[table]_[column]_active
idx_notifications_unread

-- (Priority 1)

BEGIN;

-- MEMBERSHIPS: Core relationship queries
CREATE INDEX CONCURRENTLY idx_memberships_user_id 
  ON memberships(user_id);
  
CREATE INDEX CONCURRENTLY idx_memberships_bubble_id 
  ON memberships(bubble_id);
  
CREATE INDEX CONCURRENTLY idx_memberships_user_bubble 
  ON memberships(user_id, bubble_id);

-- EVENTS: Bubble detail pages
CREATE INDEX CONCURRENTLY idx_events_bubble_id 
  ON events(bubble_id);
  
CREATE INDEX CONCURRENTLY idx_events_bubble_start 
  ON events(bubble_id, start_time DESC);
  
CREATE INDEX CONCURRENTLY idx_events_created_at 
  ON events(created_at DESC);

-- EVENT_ATTENDEES: Event detail pages
CREATE INDEX CONCURRENTLY idx_event_attendees_event_id 
  ON event_attendees(event_id);
  
CREATE INDEX CONCURRENTLY idx_event_attendees_user_id 
  ON event_attendees(user_id);
  
CREATE INDEX CONCURRENTLY idx_event_attendees_event_user 
  ON event_attendees(event_id, user_id);

-- NOTIFICATIONS: Notification feed
CREATE INDEX CONCURRENTLY idx_notifications_user_id 
  ON notifications(user_id);
  
CREATE INDEX CONCURRENTLY idx_notifications_user_created 
  ON notifications(user_id, created_at DESC);

-- BUBBLE_VISITS: Tracking/analytics
CREATE INDEX CONCURRENTLY idx_bubble_visits_user_id 
  ON bubble_visits(user_id);
  
CREATE INDEX CONCURRENTLY idx_bubble_visits_bubble_id 
  ON bubble_visits(bubble_id);
  
CREATE INDEX CONCURRENTLY idx_bubble_visits_user_bubble 
  ON bubble_visits(user_id, bubble_id);

COMMIT;
Note: CONCURRENTLY allows index creation without locking the table (safe for production).

-- BUBBLE CRITICAL INDEXES
-- (Composite Indexes:)
-- ==================================================

Rule of Thumb:
Composite index column order:
1. Equality filters (WHERE column = value)
2. Range filters (WHERE column > value)
3. Sort columns (ORDER BY column)

sql
-- Events in a Bubble, ordered by start time
CREATE INDEX idx_events_bubble_start ON events(bubble_id, start_time DESC);

-- User's notifications, ordered by recency
CREATE INDEX idx_notifications_user_created ON notifications(user_id, created_at DESC);

-- Posts in a Bubble, ordered by recency
CREATE INDEX idx_bulletin_posts_bubble_created ON bulletin_posts(bubble_id, created_at DESC);

-- Replies to a post, ordered by time
CREATE INDEX idx_bulletin_replies_post_created ON bulletin_replies(post_id, created_at DESC);

-- Memberships for a user (to check if user is in a Bubble)
CREATE INDEX idx_memberships_user_bubble ON memberships(user_id, bubble_id);

-- Attendees for an event (to check if user is attending)
CREATE INDEX idx_event_attendees_event_user ON event_attendees(event_id, user_id);
